#include<stdio.h>
int greater(int , int );
int main()
{
    int a,b;
    printf("enter two numbers :\n");
    scanf("%d%d",&a,&b);
    printf("greater number between %d and %d is %d\n",a,b,greater(a, b));
    return 0;
}
int greater(int x, int y)
{
    if (x>y)
    {
        return x;
    }
    else
    {
        return y;
    }
}
